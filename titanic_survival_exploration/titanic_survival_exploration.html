        # 
        if passenger['Sex'] == 'female':
            if passenger['Age'] > 20 and passenger['Age'] < 50 and passenger['Pclass'] == 3:
                predictions.append(0)
            else:
                if passenger['Age'] < 10 and (passenger['Pclass'] == 1 or passenger['SibSp'] >= 3):
                    predictions.append(0) 
                else:
                    predictions.append(1)
