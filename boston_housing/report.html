def performance_metric2(y_true, y_predict):
    """"""
    y_average = np.mean(y_true)
    ss_tot = 0
    for i in y_true:
        ss_tot += (i - y_average) ** 2
    
    ss_res_data = np.array(y_true) - np.array(y_predict)
    ss_res = 0
    for j in ss_res_data:
        ss_res += j ** 2
    
    score = 1 - ss_res / ss_tot

    return score
